import pygame
import random

# Define some colors
BLACK = (0, 0, 0)
WHITE = (255, 255, 255)
GREEN = (0, 255, 0)
RED = (255, 0, 0)
BLUE = (0, 0, 255)

colors = [GREEN, RED, BLUE]
lenco = len(colors)
dc = random.randint(0, lenco - 1)

pygame.init()


# Set the width and height of the screen [width, height]
size = (700, 500)
screen = pygame.display.set_mode(size)

pygame.display.set_caption("My Game")

# Loop until the user clicks the close button.
done = False

# Used to manage how fast the screen updates
clock = pygame.time.Clock()

# --- Define some variables
dx = random.randint(-10, 10)
dy = random.randint(-10, 10)
da = random.randint(-10, 10)
db = random.randint(-10, 10)
dr = random.randint(20, 80)
dr2 = random.randint(20, 80)
dw = random.randint(2, 10)
dw2 = random.randint(2, 10)
x = 350
y = 250
a = 350
b = 250


# -------- Main Program Loop -----------
while not done:
    # --- Main event loop
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            done = True
    

    # --- Looped variables here
    colors = [GREEN, RED, BLUE]
    lenco = len(colors)
    dc = random.randint(0, lenco - 1)


    
    # --- Game logic should go here

    # --- Screen-clearing code goes here

    # Here, we clear the screen to white. Don't put other drawing commands
    # above this, or they will be erased with this command.

    # If you want a background image, replace this clear with blit'ing the
    # background image.
    screen.fill(BLACK)

    # --- Drawing code should go here
    pygame.draw.circle(screen, colors[dc], (x, y), dr, dw)
    x += dx
    y += dy
    if x + dx < 30 or x + dx > 670:
        dx = 0 - dx
    if y + dy < 30 or y + dy > 470:
        dy = 0 - dy
    pygame.draw.circle(screen, colors[dc], (a, b), dr2, dw2)
    a += da
    b += db
    if a + da < 30 or a + da > 670:
        da = 0 - da
    if b + db < 30 or b + db > 470:
        db = 0 - db 
    
    pygame.draw.line(screen, colors[dc], (0, 0), (0, 500), 20)
    pygame.draw.line(screen, colors[dc], (0, 0), (700, 0), 20)
    pygame.draw.line(screen, colors[dc], (700, 500), (0, 500), 20)
    pygame.draw.line(screen, colors[dc], (700, 500), (700, 0), 20)
    
    


    # --- Go ahead and update the screen with what we've drawn.
    pygame.display.flip()

    # --- Limit to 60 frames per second
    clock.tick(60)

# Close the window and quit.
pygame.quit()
exit() # Needed when using IDLE
